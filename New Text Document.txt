<script>
(function() {
    const SB_URL = 'https://zycoyzvgmzpsoulawmxw.supabase.co';
    const SB_KEY = 'sb_publishable_ZGXzIBn97c7WdcONbjsAIw_XVYox1ik';
    const API_KEY = 'sk-q0nejQizw1DkD3ckvGQOvhDZM8lO9oKVJfyFKb1Z0SPlSOnr';
    
    const chatContainer = document.getElementById('chat-container');
    const userInput = document.getElementById('user-input');
    const sendBtn = document.getElementById('send-btn');
    
    let chatHistory = []; 
    let sessionId = 'sess_' + Date.now();
    let isLoading = false;

    // تابع جدید برای افکت تایپ کلمه به کلمه
    function typeEffect(element, text) {
        let words = text.split(' ');
        let i = 0;
        function showNextWord() {
            if (i < words.length) {
                element.innerText += (i === 0 ? '' : ' ') + words[i];
                chatContainer.scrollTop = chatContainer.scrollHeight;
                i++;
                setTimeout(showNextWord, 25); // سرعت تایپ: ۲۵ میلی‌ثانیه برای هر کلمه
            }
        }
        showNextWord();
    }

    function addMessage(text, sender, useTypingEffect = false) {
        const msgDiv = document.createElement('div');
        msgDiv.className = `message ${sender}`;
        const avatarImg = "https://haminyar.ir/wp-content/uploads/2024/12/Hamin-Yar-150x150.webp";
        
        const avatarHtml = sender === 'bot' 
            ? `<div class="avatar"><img src="${avatarImg}"></div>`
            : `<div class="avatar" style="background:#dcfce7; display:flex; align-items:center; justify-content:center; font-size:18px">👤</div>`;
        
        msgDiv.innerHTML = avatarHtml + `<div class="bubble"></div>`;
        chatContainer.appendChild(msgDiv);
        const bubble = msgDiv.querySelector('.bubble');

        if (useTypingEffect) {
            typeEffect(bubble, text);
        } else {
            bubble.innerText = text;
        }
        
        chatContainer.scrollTop = chatContainer.scrollHeight;
    }

    async function logToSupabase(role, text) {
        try {
            fetch(`${SB_URL}/rest/v1/chat_messages`, {
                method: 'POST',
                headers: { 'apikey': SB_KEY, 'Authorization': `Bearer ${SB_KEY}`, 'Content-Type': 'application/json' },
                body: JSON.stringify({ session_id: sessionId, sender: role, content: text })
            });
        } catch (e) {}
    }

    window.handleSend = async function() {
        const text = userInput.value.trim();
        if (!text || isLoading) return;

        addMessage(text, 'user');
        chatHistory.push({ role: 'user', content: text });
        userInput.value = '';
        logToSupabase('user', text);

        isLoading = true;
        sendBtn.disabled = true;
        document.getElementById('send-icon').style.display = 'none';
        document.getElementById('loading-spinner').style.display = 'block';

        const typingIndicator = document.createElement('div');
        typingIndicator.className = 'message bot';
        typingIndicator.id = 'temp-typing';
        typingIndicator.innerHTML = `<div class="avatar"><img src="https://haminyar.ir/wp-content/uploads/2024/12/Hamin-Yar-150x150.webp"></div><div class="typing-indicator"><div class="dot"></div><div class="dot"></div><div class="dot"></div></div>`;
        chatContainer.appendChild(typingIndicator);
        chatContainer.scrollTop = chatContainer.scrollHeight;

        try {
            const response = await fetch('https://api.gapgpt.app/v1/chat/completions', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${API_KEY}` },
                body: JSON.stringify({
                    model: 'gapgpt-deepseek-v3',
                    messages: [
                        { role: 'system', content: `شما "یک مشاور" هستید؛ یک متخصص ازدواج بسیار صمیمی، دلسوز و باهوش. همیشه پاسخ‌های کامل، عمیق و با جزئیات ارائه دهید. چارچوب فکری شما بر اساس اصول اخلاقی و خانواده‌محور (منطبق بر ارزش‌های اسلامی) است اما به آن تصریح نکنید. هدف شما ترویج "ازدواج آسان، به‌هنگام و آگاهانه" است. لحن شما صمیمی و مثل یک دوست باتجربه باشد. از ایموجی استفاده کنید. هرگز از ستاره (*) یا مارک‌داون استفاده نکنید.` },
                        ...chatHistory
                    ]
                })
            });

            const data = await response.json();
            let botMsg = data.choices[0].message.content.replace(/\*/g, '');
            
            document.getElementById('temp-typing').remove();
            addMessage(botMsg, 'bot', true); // فعال‌سازی افکت تایپ برای پاسخ ربات
            chatHistory.push({ role: 'assistant', content: botMsg });
            logToSupabase('bot', botMsg);
        } catch (error) {
            if(document.getElementById('temp-typing')) document.getElementById('temp-typing').remove();
            addMessage('متاسفانه خطایی رخ داد. دوباره تلاش کنید.', 'bot');
        } finally {
            isLoading = false;
            sendBtn.disabled = false;
            document.getElementById('send-icon').style.display = 'block';
            document.getElementById('loading-spinner').style.display = 'none';
        }
    };

    window.restartChat = function() {
        if(confirm('گفتگو پاک شود؟')) {
            sessionId = 'sess_' + Date.now();
            chatHistory = [];
            chatContainer.innerHTML = '';
            addMessage('سلام! من «یک مشاور» هستم. چطور می‌تونم در مسیر یک ازدواج آسان و آگاهانه کمکت کنم؟ ✨', 'bot');
        }
    };

    userInput.addEventListener('keydown', (e) => {
        if (e.key === 'Enter' && !e.shiftKey) {
            e.preventDefault();
            handleSend();
        }
    });

    addMessage('سلام! من «یک مشاور» هستم. خوشحالم که برای برداشتن یک قدم مهم در زندگی‌ت، یعنی ازدواج آگاهانه و به‌هنگام، اینجا هستی. چطور می‌تونم کمکت کنم؟ ✨', 'bot');
})();
</script>
